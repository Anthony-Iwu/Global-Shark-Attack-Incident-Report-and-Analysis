import pandas as pd
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv(r'global_shark_attacks.csv', encoding='latin1')
df['year'] = pd.to_numeric(df['year'], errors='coerce')  # Clean year column
df = df.dropna(subset=['year', 'type'])  # Drop rows with missing year/type

# Filter last 50 years and valid types
df = df[(df['year'] >= 1973) & (df['year'] <= 2023)]
df_type = df[df['type'].isin(['Unprovoked', 'Provoked'])]

# Group data
annual_trend = df_type.groupby(['year', 'type']).size().unstack().fillna(0)

# Plot
plt.figure(figsize=(12, 6))
plt.plot(annual_trend.index, annual_trend['Unprovoked'], label='Unprovoked')
plt.plot(annual_trend.index, annual_trend['Provoked'], label='Provoked')
plt.xlabel('Year')
plt.ylabel('Number of Attacks')
plt.title('Annual Shark Attacks (1973–2023)')
plt.legend()
plt.grid(True)
plt.savefig("Annual Shark Attacks.png")
plt.show()

# Seasonal patterns (monthly)
df['date'] = pd.to_datetime(df['date'], errors='coerce')
df['month'] = df['date'].dt.month
monthly = df['month'].value_counts().sort_index()

# Diurnal patterns (time of day)
def categorize_time(t):
    if pd.isna(t): return 'Unknown'
    if 'h' in str(t):
        hour = int(str(t).split('h')[0])
        if 5 <= hour < 12: return 'Morning'
        elif 12 <= hour < 17: return 'Afternoon'
        elif 17 <= hour < 21: return 'Evening'
        else: return 'Night'
    else: return 'Unknown'

# Plot
fig, axes = plt.subplots(1, 2, figsize=(15, 5))
axes[0].bar(monthly.index, monthly.values)
axes[0].set_title('Monthly Distribution of Attacks')
plt.show()

# Highlight years with high media attention (example: 2001, 2015)
media_years = [2001, 2015]
for year in media_years:

plt.text(2001, annual_counts.get(2001, 0)+5, '2001: "Summer of the Shark"', color='red')
